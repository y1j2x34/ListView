创建和销毁


    // 创建
    var explorer = $("#explorer").explorer({
        theme:"grid", // 可选
        spec:"md", // 可选
        data:[{
            cfg:{
                icon:"icon-image-file",
                name:"文字"
            }
        },...] // 可选,使用update方法
    });
    explorer.update({
        data:[{
            cfg:{
                icon:"icon-image-file",
                name:"文字"
            }
        },...]
    });
    // 销毁
    $("#explorer").explorer(false);
